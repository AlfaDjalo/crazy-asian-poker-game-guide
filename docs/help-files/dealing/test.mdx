This page tests the PokerBoardViewer.

<PokerBoardViewer
  configPath="/data/boards/single-board.json"
  playerHandSize={4}
  dealPlayerHand={true}
  renderExtra={({ playerHand, boardState }) => {
    const boardCards = boardState
      ? boardState.flat().filter(cell => cell && cell.card).map(cell => cell.card)
      : [];
    return <PokerHandEvaluator playerHand={playerHand} boardCards={boardCards} mode="high" />;
  }}
/>
